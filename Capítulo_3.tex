\chapter{Triángulos, polígonos y transformaciones}

En este capítulo se presentan las propiedades fundamentales de los triángulos y se encuentran fórmulas para hallar las coordenadas del baricentro, circuncentro, ortocentro e incentro de un triángulo dado a partir de las coordenadas de sus vértices. También se presentan los criterios de semejanza de triángulos, se estudian los triángulos rectángulos y la trigonometría, y se demuestra la fórmula de Heron para hallar el área de un triángulo a partir de las medidas de sus lados. Además, se deducen fórmulas para hallar el área de un polígono a partir de las coordenadas de sus vértices y se introduce el teorema de Pick, que proporciona una fórmula para el área de un polígono cuyos vértices tienen coordenadas enteras. 

Desde un punto de vista analítico, se muestran las transformaciones geométricas del plano, como las traslaciones, reflexiones, rotaciones y dilataciones a partir del uso de herramientas vectoriales y de matrices $2\times 2$.

\section{Triángulos}

Un triángulo, como el de la figura~\ref{fig:placeholder2.20}, está formado por tres puntos no colineales conectados por segmentos. Estos puntos no colineales se denominan vértices, y los segmentos que los unen se llaman lados.

o $\pi $ radianes.

\subsection{Coordenadas del baricentro}

Sea $T$ el triángulo con vértices $A=\left( a_{1},a_{2}\right) ,B=\left(
b_{1},b_{2}\right) $ y $C=\left( c_{1},c_{2}\right)$, como se muestra en la figura~\ref{fig:placeholder2.23}.

\begin{figure}[H]
    \centering
    \caption{Baricentro}
    \includegraphics[width=0.6\linewidth]{Images/SDY12S5W.png}
    \label{fig:placeholder2.23}
\end{figure}

Sea $L_{1}$ la mediana que pasa por $A=\left( a_{1},a_{2}\right)$ y $E=\left( \frac{b_{1}+c_{1}}{2},\frac{b_{2}+c_{2}}{2}\right)$, la pendiente
de $L_{1}$ es:
\[
m=\frac{\frac{b_{2}+c_{2}-2a_{2}}{2}}{\frac{b_{1}+c_{1}-2a_{1}}{2}}=\frac{%
b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}}
\]
Por lo tanto, la ecuación de la recta $L_{1}$ es $y=\frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}}\left( x-a_{1}\right) +a_{2}$.

Sea $L_{2}$ la mediana que pasa por $B=\left( b_{1},b_{2}\right) $ y $F=\left( \frac{a_{1}+c_{1}}{2},\frac{a_{2}+c_{2}}{2}\right)$, la pendiente
de $L_{2}$ es:
\[m=\frac{\frac{a_{2}+c_{2}-2b_{2}}{2}}{\frac{b_{1}+c_{1}-2a_{1}}{2}}=\frac{%
a_{2}+c_{2}-2b_{2}}{a_{1}+c_{1}-2b_{1}}\]
De esto que la ecuación de la recta de $L_{2}$ es$y=\frac{a_{2}+c_{2}-2b_{2}}{%
a_{1}+c_{1}-2b_{1}}\left( x-b_{1}\right) +b_{2}$.

El punto de intersección de las dos medianas ($L_{1}$ y $L_{2}$) se obtiene de la siguiente manera:
\begin{align*}
    &\frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}}\left( x-a_{1}\right) +a_{2}
    = \frac{a_{2}+c_{2}-2b_{2}}{a_{1}+c_{1}-2b_{1}}\left( x-b_{1}\right) +b_{2} \\
    &\left( \frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}}
    - \frac{a_{2}+c_{2}-2b_{2}}{a_{1}+c_{1}-2b_{1}} \right) x \\
    &\hspace{3em} =
    \frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}} a_{1}
    - \frac{a_{2}+c_{2}-2b_{2}}{a_{1}+c_{1}-2b_{1}} b_{1}
    + b_{2} - a_{2}
\end{align*}
Se resuelve para $x$:
\begin{align*}
    x &= \frac{
        \frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}}a_{1} 
        - \frac{a_{2}+c_{2}-2b_{2}}{a_{1}+c_{1}-2b_{1}}b_{1} 
        + b_{2} - a_{2}
    }{
        \frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}} 
        - \frac{a_{2}+c_{2}-2b_{2}}{a_{1}+c_{1}-2b_{1}}
    } \\[2ex]
    &= \frac{
        \left( a_{1}+b_{1}+c_{1}\right) 
        \dfrac{
            a_{1}b_{2} - a_{2}b_{1} - a_{1}c_{2} + a_{2}c_{1} + b_{1}c_{2} - b_{2}c_{1}
        }{
            2a_{1}^{2} - 5a_{1}b_{1} + a_{1}c_{1} + 2b_{1}^{2} + b_{1}c_{1} - c_{1}^{2}
        }
    }{
        3 \dfrac{
            a_{1}b_{2} - a_{2}b_{1} - a_{1}c_{2} + a_{2}c_{1} + b_{1}c_{2} - b_{2}c_{1}
        }{
            2a_{1}^{2} - 5a_{1}b_{1} + a_{1}c_{1} + 2b_{1}^{2} + b_{1}c_{1} - c_{1}^{2}
        }
    } \\[2ex]
    &= \frac{a_{1}+b_{1}+c_{1}}{3}
\end{align*}
Para obtener la coordenada $y$, reemplazamos $x=\frac{a_{1}+b_{1}+c_{1}}{3}$ en la ecuación de $L_{1}$:
\begin{align*}
y &= \frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}}\left( \frac{a_{1}+b_{1}+c_{1}}{3}-a_{1}\right) +a_{2} \\
  &= \frac{b_{2}+c_{2}-2a_{2}}{b_{1}+c_{1}-2a_{1}}\left( \frac{b_{1}-2a_{1}+c_{1}}{3}\right) +a_{2} \\
  &= \frac{b_{2}+c_{2}-2a_{2}}{3}+a_{2} \\
  &= \frac{a_{2}+b_{2}+c_{2}}{3}.
\end{align*}

Así que las coordenadas del baricentro son:
\[
\left( \overline{x},\overline{y}\right) =\left( \frac{a_{1}+b_{1}+c_{1}}{3},%
\frac{a_{2}+b_{2}+c_{2}}{3}\right)
\]

\subsubsection{Cálculo del circuncentro vectorialmente}

Sea $T$ el triángulo con vértices $A=\left( a_{1},a_{2}\right) ,B=\left(
b_{1},b_{2}\right) $ y $C=\left( c_{1},c_{2}\right)$, que se ilustra en la figura~\ref{fig:placeholder2.26}.

\begin{figure}[H]
    \centering
    \caption{Circuncentro}
    \includegraphics[width=0.4\linewidth]{Images/SDY12S5Z.png}
    \label{fig:placeholder2.26}
\end{figure}

De las propiedades del circuncentro y la figura anterior, tenemos:
\[\overrightarrow{DG}+\overrightarrow{GF}=\overrightarrow{DF}, \quad \overrightarrow{DG}\cdot \overrightarrow{AB}=0, \quad \overrightarrow{GF}\cdot \overrightarrow{AC}=0\]
Como $\overrightarrow{DG}$ es perpendicular a $\overrightarrow{AB}$, $\overrightarrow{DG}=\alpha v$, donde $v$ es perpendicular a $\overrightarrow{AB}$
\begin{align*}
    \overrightarrow{GF}\cdot \overrightarrow{AC}&=0 \\
    \left( \overrightarrow{DF}-\overrightarrow{DG}\right) \cdot \overrightarrow{AC}&=0 \\
    \left( \overrightarrow{DF}-\alpha v\right) \cdot \overrightarrow{AC}&=0 \\
    \overrightarrow{DF}\cdot \overrightarrow{AC}-\alpha v\cdot \overrightarrow{AC}&=0 
\end{align*}
Al resolver para $\alpha$ tenemos:
\begin{align*}
    \alpha &=\frac{\overrightarrow{DF}\cdot \overrightarrow{AC}}{v\cdot \overrightarrow{AC}} \quad \text{y} \quad G=\overrightarrow{OG} \\ 
    &=\overrightarrow{OD}+\overrightarrow{DG} \\ 
    &=D+\frac{\overrightarrow{DF}\cdot \overrightarrow{AC}}{v\cdot \overrightarrow{AC}}v
\end{align*}

\textbf{Fórmula para calcular el circuncentro}
\[G=D+\frac{\overrightarrow{DF}\cdot \overrightarrow{AC}}{v\cdot \overrightarrow{AC}}v\]
Donde $D=\frac{1}{2}\left( A+B\right) ,E=\frac{1}{2}\left( B+C\right) ,F=%
\frac{1}{2}\left( A+C\right) $ y $v$ es perpendicular a $\overrightarrow{AB}$.

\begin{example}
Halle el circuncentro del triángulo con vértices $A=\left( -2,2\right)$, $B=\left( 6,0\right) $ y $C=\left( 4,8\right)$.
\end{example}

\enlargethispage{\baselineskip}

\textbf{Solución}

Se hallan los puntos medios de $D=\left( 2,1\right)$, $E=\left( 5,4\right)$ y $F=\left( 1,5\right)$
\[
    \overrightarrow{DF}=\left( -1,4\right) \quad
    \overrightarrow{AC}=\left( 6,6\right) \quad
    \overrightarrow{AB}=\left( 8,-2\right) \quad
    v=\left( 2,8\right)\!\text{, perpendicular a } \overrightarrow{AB}
\]
\begin{align*}
G &= D+\frac{\overrightarrow{DF}\cdot \overrightarrow{AC}}{v\cdot \overrightarrow{AC}}v \\
  &= \left( 2,1\right) +\frac{-6+24}{12+48}\left( 2,8\right) \\
  &= \left( 2,1\right) +\frac{3}{10}\left( 2,8\right) \\
  &= \left( \frac{13}{5},\frac{17}{5}\right) \\
  &= (2.6,3.4)
\end{align*}
La gráfica del triángulo y su circuncentro se muestran en la figura~\ref{fig:placeholder2.27}.

\subsubsection{Coordenadas del ortocentro vectorialmente}

Para el triángulo con vértices $A$, $B$ y $C$, que se muesta en la figura~\ref{fig:placeholder2.30}, se tiene:

\begin{figure}[H]
    \centering
    \caption{Ortocentro del triángulo $ABC$}
    \includegraphics[width=0.4\textwidth]{Images/SDY12S67.png}
    \label{fig:placeholder2.30}
\end{figure}

$\overrightarrow{AP}+\overrightarrow{PC}=\overrightarrow{AC},$ de donde$\ 
\overrightarrow{PC}=\overrightarrow{AC}-\overrightarrow{AP}$.

De las propiedades del ortocentro:

Si $v$ es un vector perpendicular a $\overrightarrow{BC},$ como también $%
\overrightarrow{AP}$ es perpendicular a $\overrightarrow{BC},$ entonces $%
\overrightarrow{AP}=cv$.
\begin{align*}
\overrightarrow{PC} \cdot \overrightarrow{AB} &= 0 \\
\left( \overrightarrow{AC} - \overrightarrow{AP} \right) \cdot \overrightarrow{AB} &= 0 \\
\left( \overrightarrow{AC} - c v \right) \cdot \overrightarrow{AB} &= 0 \\
\overrightarrow{AC} \cdot \overrightarrow{AB} - c v \cdot \overrightarrow{AB} &= 0 \\
c v \cdot \overrightarrow{AB} &= \overrightarrow{AC} \cdot \overrightarrow{AB} \\
c &= \frac{\overrightarrow{AC} \cdot \overrightarrow{AB}}{v \cdot \overrightarrow{AB}} \\
P = \overrightarrow{OP} = \overrightarrow{OA} + \overrightarrow{AP} &= A + c v = A + \frac{\overrightarrow{AC} \cdot \overrightarrow{AB}}{v \cdot \overrightarrow{AB}} v
\end{align*}

\enlargethispage{\baselineskip}

\textbf{Fórmula para el ortocentro}
\[P=A+\frac{\overrightarrow{AC}\cdot \overrightarrow{AB}}{v\cdot 
\overrightarrow{AB}}v,\quad \text{donde } v \text{ es un vector perpendicular a } \overrightarrow{BC}
\]
Esta fórmula para el ortocentro en coordenadas queda así:

Si $A=\left( a_{1},a_{2}\right) ,B=\left( b_{1},b_{2}\right) $ y $C=\left(
c_{1},c_{2}\right)$, tenemos que un vector perpendicular a $\overrightarrow{BC}=\left(c_{1}-b_{1},c_{2}-b_{2}\right) $ es $v=\left( c_{2}-b_{2},-\left(
c_{1}-b_{1}\right) \right)$.
\begingroup\fontsize{10pt}{12pt}\selectfont
\begin{align*}
\overrightarrow{AC} &= \left( c_{1} - a_{1},\ c_{2} - a_{2} \right) \\
\overrightarrow{AB} &= \left( b_{1} - a_{1},\ b_{2} - a_{2} \right) \\
\overrightarrow{AC} \cdot \overrightarrow{AB} 
  &= \left( c_{1} - a_{1},\ c_{2} - a_{2} \right) \cdot \left( b_{1} - a_{1},\ b_{2} - a_{2} \right) \\
  &= \left( c_{1} - a_{1} \right) \left( b_{1} - a_{1} \right) + \left( c_{2} - a_{2} \right)\left( b_{2} - a_{2} \right) \\
v \cdot \overrightarrow{AB} 
  &= \left( c_{2} - b_{2},\ -\left( c_{1} - b_{1} \right) \right) \cdot \left( b_{1} - a_{1},\ b_{2} - a_{2} \right) \\
  &= \left( c_{2} - b_{2} \right) \left( b_{1} - a_{1} \right) - \left( c_{1} - b_{1} \right) \left( b_{2} - a_{2} \right) \\
P &= A + \frac{ \overrightarrow{AC} \cdot \overrightarrow{AB} }{ v \cdot \overrightarrow{AB} }\, v \\
  &= \left( a_{1}, a_{2} \right)
    + \frac{ \left( c_{1} - a_{1} \right) \left( b_{1} - a_{1} \right)
    + \left( c_{2} - a_{2} \right) \left( b_{2} - a_{2} \right) }
    { \left( c_{2} - b_{2} \right) \left( b_{1} - a_{1} \right)
    - \left( c_{1} - b_{1} \right) \left( b_{2} - a_{2} \right) }
    \left( c_{2} - b_{2},\, -\left( c_{1} - b_{1} \right) \right)
\end{align*}
\endgroup

\begin{example}
Halle el ortocentro del triángulo con vértices $A=\left( -1,-1\right)$, $B=\left( 6,0\right) $ y $C=\left( 3,5\right)$.
\end{example}

\textbf{Solución}

Aplicamos la fórmula para el ortocentro.
\begin{align*}
\overrightarrow{BC} &= \left( -3,5 \right) \\
v &= \left( 5,3 \right) \ \text{perpendicular a } \overrightarrow{BC}. \\
\overrightarrow{AC} &= \left( 4,6 \right), \quad \overrightarrow{AB} = \left( 7,1 \right) \\
P &= A + \frac{\overrightarrow{AC}\cdot \overrightarrow{AB}}{v\cdot \overrightarrow{AB}} v \\
  &= \left( -1,-1 \right) + \frac{28+6}{38} \left( 5,3 \right) \\
  &= \left( \frac{66}{19},\, \frac{32}{19} \right) \approx \left( 3.4737,\, 1.6842 \right)
\end{align*}

\textbf{Área de un triángulo usando el incentro}

\begin{figure}[H]
    \centering
    \caption{Incentro}
    \includegraphics[width=0.5\textwidth]{Images/SDY12T6B.png}
    \label{fig:placeholder2.34}
\end{figure}

Sea $ABC$ el triángulo con lados $a=BC,b=AC$ y $c=AB$, como se observa en la figura~\ref{fig:placeholder2.34}.

Sabemos que $CF=CE,AF=AD,BD=BE$ \ y \ $ID=IE=IF=r$
\begin{align*}
\text{área}\left( \triangle ABC\right) 
    &= 2\,\text{área}\left( \triangle ADI\right) + 2\,\text{área}\left( \triangle BDI\right) + 2\,\text{área}\left( \triangle CEI\right) \\
    &= \frac{1}{2}\left( 2rAD + 2rDB + 2rEC \right) \\
    &= \frac{r}{2}\left( AD + AF + DB + BE + EC + CF \right) \\
    &= \frac{r\left( a + b + c \right) }{2}
\end{align*}
Por otro lado, si $h$ es la altura perpendicular al lado $\overline{AB},$ se tiene que $\sin A=\frac{h}{AC}$, así que:\pdfcomment{se unieron dos párrafos cortos para facilitar la lectura}
\[
\text{área}\left( \triangle ABC\right) =\frac{1}{2}AB\ast h=\frac{1}{2}%
\left( AB\right) \left( AC\right) \sin A=\frac{bc\sin A}{2}
\]
Luego, se cumple que:
\[
\frac{r\left( a+b+c\right) }{2}=\frac{bc\sin A}{2}
\]

\textbf{Coordenadas del incentro}

Sea $ABC$ el triángulo con lados $a=BC=\left\Vert \overrightarrow{BC}\right\Vert$, $b=AC=\left\Vert \overrightarrow{AC}\right\Vert$ y $c=AB=\left\Vert \overrightarrow{AB}\right\Vert$, $P$ un punto sobre el segmento $\overline{AB}$, y $Q$ un punto sobre el segmento $\overline{AC}$, tales que $\overline{PI}$ sea paralelo a $\overline{AC}$ y $\overline{QI}$ sea paralelo a $\overline{AB}$, como se observa en la figura~\ref{fig:placeholder2.35}. Como $I$ es el incentro y está sobre la bisectriz del ángulo $A$, el paralelogramo $APIQ$ es un rombo. Sea $m$ la medida del lado de este rombo. El ángulo $IPD$ mide igual que el ángulo $\angle A$, así, en el triángulo rectángulo $IDP$, tenemos que $\sin A=\frac{r}{m}$.

\begin{figure}[H]
    \centering
    \caption{Coordenadas del incentro}
    \includegraphics[width=0.5\textwidth]{Images/SDY12T6C.png}
    \label{fig:placeholder2.35}
\end{figure}

\enlargethispage*{\baselineskip}

Por otro lado, del área de $\triangle ABC$ se obtiene de la igualdad $\frac{r\left( a+b+c\right) }{2}=\frac{bc\sin A}{2}$. Reemplazando $\sin A=\frac{r}{m}$, se tiene:
\[r\left( a+b+c\right) =bc\frac{r}{m} \quad \text{y} \quad m=\frac{bc}{a+b+c}\]
Consideremos los vectores unitarios en dirección de $\overrightarrow{AB}$ y $\overrightarrow{AC}$, estos son $\frac{\overrightarrow{AB}}{\left\Vert \overrightarrow{AB}\right\Vert }$ y $\frac{\overrightarrow{AC}}{\left\Vert \overrightarrow{AC}\right\Vert }$, así que:
\begin{align*}
\overrightarrow{AI} 
    &= \overrightarrow{AP}+\overrightarrow{AQ} \\ 
    &= m\frac{\overrightarrow{AB}}{\left\Vert \overrightarrow{AB}\right\Vert }+m\frac{\overrightarrow{AC}}{\left\Vert \overrightarrow{AC}\right\Vert } = m\left( \frac{\overrightarrow{AB}}{c}+\frac{\overrightarrow{AC}}{b}\right) \\
    &= \frac{bc}{a+b+c}\left( \frac{\overrightarrow{AB}}{c}+\frac{\overrightarrow{AC}}{b}\right) \\
    &= \frac{1}{a+b+c}\left( b\overrightarrow{AB}+c\overrightarrow{AC}\right)
\end{align*}

Por lo tanto, las coordenadas del incentro son:
\begin{align*}
I &= \overrightarrow{OI} \\
  &= \overrightarrow{OA} + \overrightarrow{AI} \\
  &= A + \frac{1}{a+b+c} \left( b(B-A) + c(C-A) \right) \\
  &= \frac{aA + bA + cA + bB - bA + cC - cA}{a+b+c} \\
  &= \frac{aA + bB + cC}{a+b+c}
\end{align*}

\textbf{Fórmula para las coordenadas del incentro}
\[
I=\frac{aA+bB+cC}{a+b+c}
\]

\begin{example}
Encuentre las coordenadas del incentro del triángulo con vértices $A=\left(-1,0\right)$, $B=\left( 2,3\right)$ y $C=\left( -3,4\right)$.
\end{example}

\textbf{Solución}

Aplicamos la fórmula para el incentro $I$.
\begin{align*}
a &= \left\Vert \overrightarrow{BC}\right\Vert = \left\Vert C-B\right\Vert = \left\Vert \left( -5,1\right) \right\Vert = \sqrt{26} \\
b &= \left\Vert \overrightarrow{AC}\right\Vert = \left\Vert C-A\right\Vert = \left\Vert \left( 2,4\right) \right\Vert = 2\sqrt{5} \\
c &= \left\Vert \overrightarrow{AB}\right\Vert = \left\Vert B-A\right\Vert = \left\Vert \left( 3,3\right) \right\Vert = 3\sqrt{2} \\
I &= \frac{aA+bB+cC}{a+b+c} \\
  &= \frac{1}{\sqrt{26}+2\sqrt{5}+3\sqrt{2}}\left( \sqrt{26}\left( -1,0\right)
    +2\sqrt{5}\left( 2,3\right) +3\sqrt{2}\left( -3,4\right) \right) \\
  &\approx \left( -0.64303,2.1998\right)
\end{align*}

\subsubsection{Figuras semejantes}

Dos figuras geométricas semejantes tienen la misma forma, pero no necesariamente el mismo tamaño. Abreviamos \textquotedblleft es semejante a\textquotedblright\ con el símbolo $\sim $.

\subsubsection{Polígonos semejantes}

Dos polígonos son semejantes si sus ángulos miden igual y sus lados son proporcionales. En la figura~\ref{fig:placeholder2.37}, los polígonos $ABCD$ y $EFGH$ son semejantes: \pdfcomment{se unieron dos párrafos cortos para facilitar la lectura}
\[
ABCD\sim EFGH
\]
\begin{figure}[H]
    \centering
    \caption{Poligonos semejantes}
    \includegraphics[width=0.5\textwidth]{Images/SDY12T6E.png}
    \label{fig:placeholder2.37}
\end{figure}

Esto implica que sus ángulos son congruentes, es decir, que miden igual:
\[
\angle A\cong \angle E,\qquad \angle B\cong \angle F,\qquad \angle
C\cong \angle G,\qquad \angle D\cong \angle H\]
Y sus lados son proporcionales:
\[
\frac{AB}{EF}=\frac{BC}{FG}=\frac{CD}{GH}=\frac{AD}{EH}
\]
Esta razón igual de los correspondientes lados es llamado el factor de escala de la semejanza. Así que, en esta semejanza, el factor de escala $c$ es el valor correspondiente a cualquiera de los cocientes:\pdfcomment{se unieron dos párrafos cortos para facilitar la lectura}
\[
c=\frac{AB}{EF}=\frac{BC}{FG}=\frac{CD}{GH}=\frac{AD}{EH}
\]

De las medidas del dibujo, se tiene:
\[\angle A\cong \angle E,\qquad \angle B\cong \angle F,\qquad  \angle

\[\underset{\frac{18}{54}}{\frac{AB}{EF}}=\underset{\frac{14}{42}}{\frac{BC}{FG}}=\underset{\frac{8}{24}}{\frac{CD}{GH}}=\underset{\frac{16}{48}}{\frac{AD}{EH}}\]
Asimismo, los lados son proporcionales. Por lo tanto, los polígonos son proporcionales: $ABCD\sim EFGH$.

Como la altura a la hipotenusa $BD$, que se muestra en la figura~\ref{fig:placeholder2.42}, divide el triángulo $ABC$ en dos triángulos semejantes a él, tenemos:

$\triangle ADB\sim \triangle ABC$, así que sus correspondientes lados son proporcionales. Luego:

$\frac{AB}{AD}=\frac{AC}{AB}$, entonces \thinspace $E_{1}:\left( AB\right)
^{2}=\left( AC\right) \left( AD\right)$.

$\triangle CDB\sim \triangle CBA$, por lo cual sus correspondientes lados son proporcionales. Luego:

$\frac{BC}{DC}=\frac{AC}{BC}$, entonces $E_{2}:\left( BC\right) ^{2}=\left(
AC\right) \left( DC\right) $.

Al sumar las ecuaciones $E_{1}+E_{2}:\left( AB\right) ^{2}+\left( BC\right)
^{2}=\left( AC\right) \left( AD\right) +\left( AC\right) \left( DC\right) $

y $\left( AB\right) ^{2}+\left( BC\right) ^{2}=\left( AC\right) \left(
AD+DC\right) $

Por lo tanto, \ $\left( AB\right) ^{2}+\left( BC\right) ^{2}=\left( AC\right)
^{2}$ y queda demostrado el teorema.

\bigskip

Una tripla pitagórica es una terna $(x,y,z)$ de números enteros positivos, tales que $x^{2}+y^{2}=z^{2}.$

Las triplas pitagóricas son de la forma $(m^{2}-n^{2},2mn,m^{2}+n^{2})$, donde $m,n $ son enteros con $m>n$, pues se satisface la identidad pitagórica:
\[\left( m^{2}-n^{2}\right) ^{2}+\left( 2mn\right) ^{2}=\left(
m^{2}+n^{2}\right) ^{2}\]
El recíproco del teorema de Pitágoras es válido. Sean $a,b,c$ los lados de un triángulo donde $c$ es el lado más grande, se cumple:\pdfcomment{Se juntaron dos párrafos cortos en uno.}
\begin{enumerate}
    \item Si $c^{2}=a^{2}+b^{2},$ el triángulo es rectángulo.
    \item Si $c^{2}>a^{2}+b^{2},$ el triángulo es obtusángulo.
    \item Si $c^{2}<a^{2}+b^{2},$ el triángulo es acutángulo.
\end{enumerate}

\subsection{Área de un triángulo a partir de las longitudes de sus lados}

\subsubsection{Fórmula de Herón}

\enlargethispage*{\baselineskip}

Dado un triángulo $ABC$, con lados $a$,$b$ y $c$ dados, la fórmula usual para el área del triángulo, que es la mitad de la base por la altura, no permite calcular el área directamente a partir de las longitudes de los lados. 

Sin embargo, existe una fórmula para calcular el área del triángulo usando únicamente las longitudes de los lados. Esta es la llamada fórmula de Herón, que probaremos a continuación.

Llamemos $u$ a una parte de la base $b$ del triángulo, determinada por la intersección de esta con la altura, de forma que la otra parte será $b-u$, como en la figura~\ref{fig:placeholder2.53}.

\begin{figure}[H]
    \centering
    \caption{Triángulo con una altura}
    \includegraphics[width=0.5\textwidth]{Images/SDY12T73.png}
    \label{fig:placeholder2.53}
\end{figure}

Al aplicar el teorema de Pitágoras a los dos triángulos rectángulos determinados por la altura $h$, de la figura~\ref{fig:placeholder2.53}, se obtiene:
\[h^{2}=c^{2}-u^{2} \quad \text{y} \quad h^{2}=a^{2}-\left( b-u\right) ^{2}\]
Al igualar $h^{2}$:
\begin{align*}
a^{2}-\left( b-u\right) ^{2} &= c^{2}-u^{2} \\
a^{2}-b^{2}+2bu-u^{2} &= c^{2}-u^{2} \\
u &= \frac{b^{2}+c^{2}-a^{2}}{2b}
\end{align*}
Si se reemplaza en $h^{2}=c^{2}-u^{2}$, se obtiene:
\begin{align*}
h^{2} 
&= c^{2} - \left( \frac{(b^{2} + c^{2} - a^{2})^{2}}{2b} \right)^{2} \\
&= \frac{4b^{2}c^{2} - (b^{2} + c^{2} - a^{2})^{2}}{4b^{2}} \\
&= \frac{(2bc + b^{2} + c^{2} - a^{2}) (2bc - b^{2} - c^{2} + a^{2})}{4b^{2}} \\
&= \frac{((b+c)^{2} - a^{2})(a^{2} - (b-c)^{2})}{4b^{2}} \\
&= \frac{(b+c+a)(b+c-a)(a+b-c)(a-b+c)}{4b^{2}}
\end{align*}
De donde $h=\frac{\sqrt{\left( b+c+a\right) \left( b+c-a\right) \left(
a+b-c\right) \left( a-b+c\right) }}{2b}$ y el área del triángulo es:
\begin{align*}
A &= \frac{1}{2}bh = \frac{1}{2}b\frac{\sqrt{(b+c+a)(b+c-a)(a+b-c)(a-b+c)}}{2b} \\[1em]
A &= \sqrt{\frac{(a+b+c)}{2}\frac{(b+c-a)}{2}\frac{(a+b-c)}{2}\frac{(a-b+c)}{2}} \\[1em]
A &= \sqrt{\frac{(a+b+c)}{2} \left( \frac{a+b+c}{2} - a \right)
      \left( \frac{a+b+c}{2} - b \right)
      \left( \frac{a+b+c}{2} - c \right) }
\end{align*}
Si llamamos $s=\frac{a+b+c}{2}$ \ al semiperímetro del triángulo, tenemos:
\[
A=\sqrt{s\left( s-a\right) \left( s-b\right) \left( s-c\right) }
\]
Esta es la conocida fórmula de Herón.

\begin{example}
Halle el área del triángulo cuyos lados miden $5$, $12$ y $13$ centímetros.
\end{example}

\textbf{Solución}

El perímetro es $P=5+12+13=30$ y el semiperímetro es $s=15$. El área del triángulo es
\[A=\sqrt{15\left( 15-5\right) \left( 15-12\right) \left( 15-13\right) }=30\,\text{cm}^{2}\]
Por otro lado, note que este triángulo es rectángulo, ya que $13^{2}=12^{2}+5^{2}$. De esto se deduce que el área es la mitad de la base ($12$) por la altura ($5$), lo cual indica que el área del triángulo es $30$ cm$^{2}.$

\section{Polígonos}

Una figura es un polígono si cumple las siguientes condiciones:
\begin{enumerate}
    \item Es una figura plana formada por tres o más segmentos lineales, llamados lados.
    \item Los lados que tienen un punto final común son no colineales.
    \item Cada lado interseca exactamente a otros dos lados, pero únicamente en los puntos finales.
\end{enumerate}

Un polígono es convexo si cada segmento que une dos puntos interiores está totalmente contenido en el polígono. Un polígono es cóncavo si no es convexo. Esto se ilustra en la figura~\ref{fig:placeholder2.54}.

La condición de que un cuadrilátero sea un paralelogramo es equivalente a cada una de las siguientes:

    \item Sus ángulos consecutivos son suplementarios.
    \item Sus diagonales se bisecan entre sí.
\end{enumerate}

En $\left\vert \det \left[ u\text{ }v\right] \right\vert $, los vectores $u$ y $v$ se escriben en columna.

\textbf{Demostración gráfica}

Supongamos que\ $u=\left( a,b\right) $ y $v=\left( c,d\right) $

\begin{figure}[H]
    \centering
    \caption{Rectángulo con paralelogramo interno}
    \includegraphics[width=0.5\linewidth]{Images/SDY12T7I.png}
    \label{fig:placeholder2.59}
\end{figure}

A partir de la figura~\ref{fig:placeholder2.59}, se obtiene:
\begin{align*}
\area\left(P\right) 
    &= \area\left(\text{Rectángulo}\right)
      - \area\left(T_1\right) - \area\left(T_2\right)
      - \area\left(T_3\right) - \area\left(T_4\right) \\
      & \quad - \area\left(R_1\right) - \area\left(R_2\right) \\
    &= (a + c)(b + d)
       - \frac{1}{2}ab - \frac{1}{2}cd
       - \frac{1}{2}ab - \frac{1}{2}cd
       - bc - bc \\
    &= ab + ad + cb + cd - ab - cd - 2bc \\
    &= ad - bc \\
    &= \det\begin{pmatrix}
        a & c \\
        b & d
      \end{pmatrix} \\
    &= \left| \det \left[ u\; v\right] \right| \\
\end{align*}

\begin{example}
Halle el área del paralelogramo generado por los vectores.
\end{example}

\textbf{Solución}
\[u=\left( 
\begin{array}{c}
15 \\ 
25%
\end{array}%
\right) \quad \text{y} \quad v=\left( 
\begin{array}{c}
22 \\ 
34%
\end{array}%
\right)\]
\begin{align*}
\area\left( P\right) &= \left\vert \det \left[ u\text{ }v\right] \right\vert \\
&= \left\vert \det \left( 
\begin{array}{cc}
15 & 22 \\ 
25 & 34
\end{array}
\right) \right\vert \\ 
&= \left\vert 15\left( 34\right) -22\left( 25\right)
\right\vert \\
&= \left\vert -40\right\vert = 40
\end{align*}

Expresamos el área del trapecio como la suma de áreas de dos triángulos y un rectángulo, como se muestra en la figura~\ref{fig:placeholder2.63}:
\begin{align*}
\area\left(ABCD\right) 
    &= \area\left( \triangle AED\right) 
      + \area\left( \triangle BCF\right)
      + \area\left( \square ABFE \right) \\
        &\quad\enspace \text{\small note que $AE = BF$} \\
    &= \frac{1}{2}(DE)(AE) + \frac{1}{2}(FC)(BF) + (EF)(AE) \\
    &= \frac{1}{2}(AE)\left( DE + 2EF + FC \right) \\
    &= \frac{1}{2}(AE)\left( EF + DE + EF + FC \right) \\
    &= \frac{1}{2}(AE)\left( EF + DC \right)
       \quad \text{\small note que $EF = AB$} \\
    &= \frac{1}{2}(AE)\left( AB + DC \right)
\end{align*}
Por lo tanto, el área de un trapecio es la mitad de la altura por la suma de sus bases.

\subsubsection{Área de un polígono con coordenadas enteras}

El teorema de Pick brinda una fórmula simple para calcular el área de un polígono cuyos vértices tienen coordenadas enteras. Fue demostrado por primera vez por Georg Alexander Pick en 1899 y es un resultado clásico de geometría. 

Un punto interior es un punto de la red que está dentro del polígono, mientras que un punto de frontera es un punto de la red que se encuentra en el límite del polígono.

\textbf{Teorema de Pick} 

Sea $A$ el área de un polígono con vértices de coordenadas enteras, como el que se muestra en la figura~\ref{fig:placeholder2.69}, $I$ el número de puntos de la red en su interior y $B$ el número de puntos en la frontera de la red, incluidos los vértices. Entonces: $A=I+\frac{1}{2}B-1$.

\begin{figure}[H]
    \centering
    \caption{Teorema de Pick}
    \includegraphics[width=0.6\linewidth]{Images/SDY12T80.png}
    \label{fig:placeholder2.69}
\end{figure}

A continuación, se presenta una demostración que utiliza un método de doble conteo basado en una triangulación del polígono. Para esto, contamos los ángulos dentro de los triángulos de dos maneras diferentes.

Un triángulo elemental es un triángulo cuyos vértices tienen coordenadas enteras y que no contiene más puntos, ni en el borde ni en el interior. Estos triángulos son aquellos que tienen vértices de coordenadas enteras y un área mínima.

Primero, enunciamos algunos resultados previos.

Un triángulo con coordenadas enteras es elemental si no tiene otros puntos con coordenadas enteras en su frontera, aparte de los vértices. Es decir, los triángulos elementales son aquellos con vértices enteros y área mínima.

El área de cualquier triángulo elemental en la red $\mathbb{Z}^{2}$ es $\frac{1}{2}$, pues el área del triángulo que une los puntos con coordenadas
enteras $O=\left( 0,0\right)$.
\[A=\left( a,b\right),\; B=\left( c,d\right) 
\quad \text{es} \quad A=\frac{1}{2}\left\vert 
\begin{array}{cc}
a & c \\ 
b & d%
\end{array}
\right\vert =\frac{1}{2}\left( ad-bc\right)
\]
Para que uno de estos triángulos tenga área mínima, se debe cumplir $ad-bc=1$. Luego, estos triángulos de área mínima tienen área $\frac{1}{2}$.

Cualquier polígono con coordenadas enteras, como el de la figura~\ref{fig:placeholder2.70}, puede ser triangulado por triángulos elementales.

\begin{figure}[H]

    \caption{Polígono triangulado}

\end{figure}

Como se ve en la figura~\ref{fig:placeholder2.71}, la suma de los ángulos externos de un polígono es 360$\degree=2\pi $ radianes.

En cada vértice del polígono, se cumple que la suma del ángulo interno con el externo es $\pi $ radianes.

\bigskip

\textbf{Demostración del teorema de Pick}

Comenzamos por dividir el polígono $P$ en $N$ triángulos elementales. Ahora, sumamos los ángulos internos de todos estos triángulos de dos maneras diferentes. Por un lado, la suma de los ángulos de cualquier triángulo es $\pi$, por lo que la suma de todos los ángulos es $S=N\pi$.

Por otra parte, en cada punto interior $i$, los ángulos de los triángulos elementales que se encuentran en $i$ suman $2\pi$. En cada punto de la frontera $b$ que no es un vértice, los ángulos de los triángulos elementales
que se encuentran en $b$ suman $\pi$. En los vértices, los ángulos no suman $\pi$, pero si sumamos los ángulos interiores en todos los vértices, obtenemos $k\pi -2\pi $, donde $k$ es el número de vértices, ya que la suma de los ángulos exteriores es $2\pi$. Este resultado se puede justificar observando que, al recorrer el perímetro del polígono, se completa una vuelta completa, es decir $2\pi $. Cabe señalar que algunos ángulos exteriores aportan un término positivo (en los vértices donde el polígono es convexo) y otros un término negativo (en los vértices donde el polígono es cóncavo).

Sea $I$ el número de puntos interiores y $B$ el número de puntos en la frontera. En total, la suma de los ángulos en los puntos de la frontera es $B\pi -2\pi $, y la suma de los ángulos en los puntos internos es $I\left( 2\pi \right) $. Por lo tanto,
\begin{equation*}
S=I\left( 2\pi \right) +B\pi -2\pi
\end{equation*}
Concluimos que
\begin{equation*}
N\pi =I\left( 2\pi \right) +B\pi -2\pi
\end{equation*}
Por lo que cancelando $\pi $ obtenemos
\begin{equation*}
N=2I+B-2
\end{equation*}

\end{equation*}

El volumen $V$ de un tetraedro, que es una pirámide de base triangular, es igual a la tercera parte del área de la base multiplicada por la altura:
\begin{equation*}
    V=\frac{1}{3}A_{b}h
\end{equation*}

\begin{figure}[H]

\begin{align*}
V &= \frac{1}{3}\left( \frac{1}{2}\left\Vert X\times Y\right\Vert \left\Vert \text{Proy}_{X\times Y}Z\right\Vert \right) \\
  &= \frac{1}{6}\left( \left\Vert X\times Y\right\Vert \left\Vert \frac{\left( X\times Y\right) \cdot Z}{\left\Vert X\times Y\right\Vert ^{2}}X\times Y\right\Vert \right) \\
  &= \frac{1}{6}\left\vert \left( X\times Y\right) \cdot Z\right\vert \\
  &= \frac{1}{6}\left\vert \det \left[ 
      \begin{array}{c}
        X \\ 
        Y \\ 
        Z
      \end{array}
    \right] \right\vert \\
  &= \frac{1}{6}\left\vert \det \left( 
      \begin{array}{ccc}
        x_{1} & x_{2} & x_{3} \\ 
        y_{1} & y_{2} & y_{3} \\ 
        z_{1} & z_{2} & z_{3}

    \right) \right\vert \\

    \right\vert \\

\subsection{Transformaciones rígidas}

Una transformación de una figura en un plano es un cambio en la posición, forma o tamaño de la figura.

En una transformación, la figura original es la preimagen y la figura resultante es la imagen.

Desde un punto de vista analítico, una transformación es una función $T:\mathbb{R}^{2}\rightarrow \mathbb{R}^{2}$.

Si $P=\left( x,y\right)$ es un punto del plano, su imagen es $P^{\prime}=T\left( P\right) =T\left( x,y\right)$.

Si $F\subseteq \mathbb{R}^{2}$ es la preimagen, $T\left( F\right) $ es la imagen. También podemos denotar como $F^{\prime}$ la imagen de $F$. Esto se ilustra en la figura~\ref{fig:placeholder3.9}.

Una isometría es una transformación en la cual la preimagen y la imagen son congruentes. Esto significa que las medidas de los ángulos y las longitudes de los lados se mantienen iguales en la preimagen y la imagen.

Una isometría también se denomina transformación rígida, porque la forma original y el tamaño de la figura no cambian. 

En esta sección, se estudian tres isometrías básicas o transformaciones rígidas: traslaciones, reflexiones y rotaciones (o giros).

Una transformación $T:\mathbb{R}^{2}\rightarrow \mathbb{R}^{2}$ es una transformación matricial si existe una matriz $2\times 2,$ $A=\left( 
\begin{array}{cc}
a & b \\ 
c & d%
\end{array}%
\right) $ tal que $T\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) =\left( 
\begin{array}{cc}
a & b \\ 
c & d%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right)$.

\subsubsection{Traslaciones}

Una traslación es una transformación sobre el plano, que desplaza todos los puntos de una figura a la misma distancia y en la misma dirección.

Para dibujar la traslación de una figura, se pueden representar en el mismo plano la preimagen y la imagen, como se ve en la figura~\ref{fig:placeholder3.11}.

La preimagen es el triángulo $ABC$, y la imagen, tras la traslación, es el triángulo $A^{\prime }B^{\prime }C^{\prime}$.

\begin{figure}[H]
    \centering
    \caption{Traslación de una figura plana}
    \includegraphics[width=0.8\linewidth]{Images/SDY12T8E.png}
    \label{fig:placeholder3.11}
\end{figure}

Analíticamente una traslación es una transformación $T:\mathbb{R}^{2}\rightarrow \mathbb{R}^{2}$ que mueve cada punto del plano $u$ según un vector dado $v$, llamado el vector de traslación.
\[T\left( u\right) =u+v\]
En coordenadas, si $u=\left( x,y\right) $ y $v=\left( a,b\right) $, entonces
\[T\left( x,y\right) =\left( x,y\right) +\left( a,b\right) =\left(
x+a,y+b\right)\]
Se cumple que las distancias entre cada punto de la imagen y su correspondiente punto en la imagen miden igual:
\begin{equation*}
d\left( A,A^{\prime }\right) =d\left( B,B^{\prime }\right)
\end{equation*}
Sean $A=\left( x_{1},y_{1}\right)$, $B=\left( x_{2},y_{2}\right)$
\begin{align*} 
    d\left( A,A^{\prime} \right) &=d\left( \left( x_{1},y_{1}\right)\!,\left(
    x_{1}+a,y_{1}+b\right) \right) =\sqrt{a^{2}+b^{2}} \\
    d\left( B,B^{\prime} \right) &=d\left( \left( x_{2},y_{2}\right)\!,\left(
    x_{2}+a,y_{2}+b\right) \right) =\sqrt{a^{2}+b^{2}}
\end{align*}
Por tanto $d\left( A,A^{\prime }\right) =d\left( B,B^{\prime }\right)$.

Las traslaciones son isometrías:

Sea $T$ la traslación $T\left( x,y\right) =\left( x,y\right) +\left(
a,b\right)$

$T$ preserva longitudes:\pdfcomment{Esta ecuación no tenía el signo de raiz cuadrada en la segunda línea, pero por contexto parece que faltaba}
\begin{align*}
d\left( T\left( x_{1},y_{1}\right),\, T\left( x_{2},y_{2}\right) \right) 
  &= d\left( \left( x_{1}+a,\, y_{1}+b \right),\; \left( x_{2}+a,\, y_{2}+b \right) \right) \\
  &= \sqrt{(x_{1}-x_{2})^2 + (y_{1}-y_{2})^2} \\
  &= d\left( (x_{1},y_{1}),\, (x_{2},y_{2}) \right)
\end{align*}
$T$ preserva ángulos:

Sean $A,B$ y $C$ puntos. Si $A^{\prime }=T\left( A\right) $, $B^{\prime
}=T\left( B\right) $ y $C^{\prime }=T\left( C\right)$, como se muestra en la figura~\ref{fig:placeholder3.12}, 

\begin{figure}[H]
    \centering
    \caption{Traslación}
    \includegraphics[width=0.8\textwidth]{Images/SDY12T8F.png}
    \label{fig:placeholder3.12}
\end{figure}

\begin{align*}
B^{\prime} - A^{\prime} 
    &= \left( x_{2} + a,\, y_{2} + b \right) - \left( x_{1} + a,\, y_{1} + b \right) \\
    &= \left( x_{2},\, y_{2} \right) - \left( x_{1},\, y_{1} \right) \\
    &= B - A. \\
C^{\prime} - A^{\prime} 
    &= \left( x_{3} + a,\, y_{3} + b \right) - \left( x_{1} + a,\, y_{1} + b \right) \\
    &= \left( x_{3},\, y_{3} \right) - \left( x_{1},\, y_{1} \right) \\
    &= C - A.
\end{align*}
Así, $\overrightarrow{A^{\prime }B^{\prime }}\cdot \overrightarrow{A^{\prime
}C^{\prime }}=\left( B^{\prime }-A^{\prime }\right) \cdot \left( C^{\prime
}-A^{\prime }\right) =\left( B-A\right) \cdot \left( C-A\right) =%
\overrightarrow{AB}\cdot \overrightarrow{AC}$

Por tanto, $\frac{\overrightarrow{A^{\prime }B^{\prime }}\cdot 
\overrightarrow{A^{\prime }C^{\prime }}}{d\left( A^{\prime },B^{\prime
}\right) d\left( A^{\prime },C^{\prime }\right) }=\frac{\overrightarrow{AB}%
\cdot \overrightarrow{AC}}{d\left( A,B\right) d\left( A,C\right) }$

Así que $\angle \left( \overrightarrow{A^{\prime }B^{\prime }},%
\overrightarrow{A^{\prime }C^{\prime }}\right) \cong \angle \left( 
\overrightarrow{AB},\overrightarrow{AC}\right)$

\textbf{Composición de traslaciones}

Sean $T_{1}\left( x,y\right) =\left( x,y\right) +\left( a,b\right) $ y $
T_{2}\left( x,y\right) =\left( x,y\right) +\left( c,d\right) $ dos traslaciones.

La composición de $T_{1}$ y $T_{2}$ nuevamente es una traslación y conmuta.
\begin{align*}
\left( T_{1}\circ T_{2}\right) (x, y)
    &= T_{1}\left( T_{2}(x, y) \right) \\
    &= T_{1}\left( (x, y) + (c, d) \right) \\
    &= T_{1}\left( x + c,\, y + d \right) \\
    &= (x + c,\, y + d) + (a, b) \\
    &= (x + c + a,\, y + d + b) \\
    &= (x, y) + (a + c,\, b + d)
\end{align*}
es la traslación según el vector suma $\left( a,b\right) +\left( c,d\right)$
\begin{align*}
\left( T_{2} \circ T_{1} \right) (x, y) 
    &= T_{2} \left( T_{1}(x, y) \right) \\
    &= T_{2} \left( (x, y) + (a, b) \right) \\
    &= (x, y) + (a, b) + (c, d) \\
    &= \left( T_{1} \circ T_{2} \right) (x, y)
\end{align*}
Luego, la composición de traslaciones es conmutativa.

\textbf{Inversa de una traslación y traslación de curvas}

Si $T\left( x,y\right) =\left( x^{\prime},y^{\prime}\right) =T\left( x,y\right) +\left( a,b\right) =\allowbreak \left( a+x,b+y\right)$, para despejar $\left( x,y\right) $ a partir de $\left( x^{\prime},y^{\prime}\right)$ se tiene:
\[\left( x,y\right) =\left( x^{\prime}-a,y^{\prime}-b\right) \]
Así que $T^{-1}\left( x,y\right) =\left( x-a,y-b\right) $

Sea $C$ una curva con ecuación implícita $F\left( x,y\right) =0$. Al trasladar $C$ $a$ unidades a la derecha y $b$ unidades arriba, la ecuación de la curva trasladada $C^{\prime }$ es $F\left( x-a,y-b\right) =0$.

\begin{example}
Sea $C$ la circunferencia con ecuación $x^{2}+y^{2}=4$. Encuentre la ecuación de la curva $C^{\prime }$ obtenida al trasladarla $3$ unidades a la derecha y $2$ unidades hacia arriba. Grafique $C$ y $C^{\prime }$.
\end{example}

\textbf{Solución}

En la circunferencia con ecuación $x^{2}+y^{2}=4$, reemplazamos $x$ por $x-3$
y $y$ por $y-2$; así se obtiene $C^{\prime }$, que es la circunferencia con ecuación: $\left( x-3\right) ^{2}+\left( y-2\right) ^{2}=4$.

La reflexión respecto a una recta $l$ es una transformación con las siguientes dos propiedades:

\begin{enumerate}
    \item Cada punto $A$ sobre la recta $l$ es invariante por la reflexión, es decir, $A^{\prime }=A$ o $T\left( A\right) =A$.
    \item Si $B$ no está sobre $l$, entonces $l$ es el bisector perpendicular del segmento $BB^{\prime }.$ Es decir, $l$ es perpendicular al segmento $BB^{\prime }$ y la distancia de $B$ a $l$ es igual a la distancia de $B^{\prime }$ a $l$, como se ilustra en la figura~\ref{fig:placeholder3.15}.
\end{enumerate}

\textbf{Reflexión respecto a la recta \boldmath{$y=mx$}}

Recta de reflexión $y=mx$ 

Vector director de la recta $v=\left( 1,m\right) $

De acuerdo a la figura~\ref{fig:placeholder3.16}, se tiene:
\begin{align*}
u &= \left( x,y\right) , \quad u^{\prime} = \left( x^{\prime}, y^{\prime} \right) \\
u^{\prime} &= T\left( u \right) = \Proy_{v}u + \Proy_{v}u - u \\
           &= 2 \Proy_{v}u - u \\
           &= 2 \frac{u\cdot v}{\left\Vert v \right\Vert ^2} v - u \\
           &= 2 \frac{x + m y}{1 + m^2} \left( 1, m \right) - \left( x, y \right) \\
           &= \frac{1}{m^2 + 1} \left( -x m^2 + 2 y m + x,\; y m^2 + 2 x m - y \right)
\end{align*}
Matricialmente, la reflexión queda:
\begin{align*}
T\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) & =\frac{1}{m^{2}+1}\left( 
\begin{array}{c}
\left( 1-m^{2}\right) x+2ym \\ 
2xm+y\left( m^{2}-1\right)%
\end{array}%
\right) \\ 
& =\frac{1}{m^{2}+1}\left( 
\begin{array}{cc}
1-m^{2} & 2m \\ 
2m & m^{2}-1%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right)
\end{align*}
Note que los números $2m,m^{2}-1$ y $m^{2}+1$ forman una tripla pitagórica.

La reflexión respecto al eje $x$ se obtiene reemplazando $m$ por $0$:
\[T\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) =\left( 
\begin{array}{cc}
1 & 0 \\ 
0 & -1%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right)\]
La reflexión respecto al eje $y$ se obtiene tomando el límite cuando $m$ tiende a $\infty$:
\begin{align*}
T\left( 
\begin{array}{c}
x \\ 
y
\end{array}
\right) 
&=
\left( 
\begin{array}{cc}
\lim\limits_{m\rightarrow \infty} \dfrac{1 - m^{2}}{m^{2} + 1} & \lim\limits_{m\rightarrow \infty} \dfrac{2m}{m^{2} + 1} \\[1em]
\lim\limits_{m\rightarrow \infty} \dfrac{2m}{m^{2} + 1} & \lim\limits_{m\rightarrow \infty} \dfrac{m^{2} - 1}{m^{2} + 1}
\end{array}
\right)
\left( 
\begin{array}{c}
x \\ 
y
\end{array}
\right) \\[1em]
&=
\left( 
\begin{array}{cc}
-1 & 0 \\ 
0 & 1
\end{array}
\right)
\left( 
\begin{array}{c}
x \\ 
y
\end{array}
\right)
\end{align*}

\paragraph{Rotaciones alrededor de un punto}

Una rotación de un ángulo $\alpha $ alrededor de un punto $R$ (llamado el \emph{centro de la rotación}) es una transformación que satisface las condiciones:
\begin{enumerate}
    \item El centro $R$ es invariante bajo la rotación, es decir, $R^{\prime }=R$.
    \item Para cada punto $A\neq R$, se cumple que $RA=R^{\prime }A$ y $m\angle ARA^{\prime}=\alpha$.
\end{enumerate}

\paragraph{Rotaciones centradas en el origen}

Una rotación alrededor de un punto es una isometría.

Transformación de rotación de un ángulo $\phi $ en sentido antihorario centrada en el origen.

\begin{figure}[H]
    \centering
    \caption{Rotación centrada en el origen}
    \includegraphics[width=0.4\textwidth]{Images/SDY12T8M.png}
    \label{fig:placeholder3.19}
\end{figure}

De  la figura~\ref{fig:placeholder3.19} tenemos:
\begin{align*}
    T\left( P \right) &= P', \quad P = (x, y), \quad P' = (x', y') \\ 
        x &= r \cos \theta, \quad y = r \sin \theta \\
        x' &= r \cos(\theta + \phi) = r \cos \theta \cos \phi - r \sin \theta \sin \phi \\
        y' &= r \sin(\theta + \phi) = r \sin \theta \cos \phi + r \cos \theta \sin \phi \\
        x' &= x \cos \phi - y \sin \phi \\
        y' &= y \cos \phi + x \sin \phi \\[1ex]
    T\left( P \right) &= T\left( 
        \begin{array}{c}
            x \\
            y
        \end{array}
    \right) = \left(
        \begin{array}{c}
            x' \\
            y'
        \end{array}
    \right) \\
    &= \left(
        \begin{array}{c}
            x \cos \phi - y \sin \phi \\
            x \sin \phi + y \cos \phi
        \end{array}
    \right) \\
    &= 
    \left(
        \begin{array}{cc}
            \cos \phi & -\sin \phi \\
            \sin \phi & \cos \phi
        \end{array}
    \right)
    \left(
        \begin{array}{c}
            x \\
            y
        \end{array}
    \right)
\end{align*}
El determinante de la matriz de rotación es:
\[\det \left( 
\begin{array}{cc}
\cos \phi & -\sin \phi \\ 
\sin \phi & \cos \phi%
\end{array}%
\right) =\cos ^{2}\phi -\left( -\sin ^{2}\phi \right) =1\]
La inversa de la matriz de rotación es:
\begin{align*}
\left(
\begin{array}{cc}
\cos \phi & -\sin \phi \\
\sin \phi & \cos \phi
\end{array}
\right)^{-1}
&= \frac{1}{\det \left(
\begin{array}{cc}
\cos \phi & -\sin \phi \\
\sin \phi & \cos \phi
\end{array}
\right)}
\left(
\begin{array}{cc}
\cos \phi & \sin \phi \\
-\sin \phi & \cos \phi
\end{array}
\right) \\
&= \left(
\begin{array}{cc}
\cos \phi & \sin \phi \\
-\sin \phi & \cos \phi
\end{array}
\right)
\end{align*}
De manera que, al despejar $\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) $ con la matriz inversa, tenemos:
\[\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) =\left( 
\begin{array}{cc}
\cos \phi & -\sin \phi \\ 
\sin \phi & \cos \phi%
\end{array}%
\right) ^{-1}\left( 
\begin{array}{c}
x^{\prime} \\ 
y^{\prime}%
\end{array}%
\right) =\left( 
\begin{array}{cc}
\cos \phi & \sin \phi \\ 
-\sin \phi & \cos \phi%
\end{array}%
\right) \left( 
\begin{array}{c}
x^{\prime} \\ 
y^{\prime}%
\end{array}%
\right)\]

\paragraph{Composición de dos rotaciones}

A continuación, probaremos que la composición de dos rotaciones centradas en el origen es una rotación centrada en el origen.

Sean \[R_{\phi }\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) =\left( 
\begin{array}{cc}
\cos \phi & -\sin \phi \\ 
\sin \phi & \cos \phi%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right)\] la rotación de un ángulo $\phi$, y \[R_{\theta }\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) =\left( 
\begin{array}{cc}
\cos \theta & -\sin \theta \\ 
\sin \theta & \cos \theta%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right)\] la rotación de un ángulo $\theta$, la composición de estas dos rotaciones es:
\begin{align*}
R_{\phi }&\left( R_{\theta }\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) \right) \\
&= R_{\phi }\left( \left( 
\begin{array}{cc}
\cos \theta & -\sin \theta \\ 
\sin \theta & \cos \theta%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) \right) \\
&= \left( 
\begin{array}{cc}
\cos \phi & -\sin \phi \\ 
\sin \phi & \cos \phi%
\end{array}%
\right) \left( 
\begin{array}{cc}
\cos \theta & -\sin \theta \\ 
\sin \theta & \cos \theta%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) \\
&= \left( 
\begin{array}{cc}
\cos \phi \cos \theta -\sin \phi \sin \theta & -\sin \theta \cos \phi -\sin
\phi \cos \theta \\ 
\sin \phi \cos \theta +\sin \theta \cos \phi & -\sin \phi \sin \theta +\cos
\phi \cos \theta%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right)
\end{align*}
Al usar las identidades de la suma para el seno y el coseno, se tiene:
\begin{align*}
&=\left( 
\begin{array}{cc}
\cos \left( \theta +\phi \right) & -\sin \left( \theta +\phi \right) \\ 
\sin \left( \theta +\phi \right) & \cos \left( \theta +\phi \right)%
\end{array}%
\right) \\
&=R_{\theta +\phi } \\
&=R_{\theta }\left( R_{\phi }\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) \right)
\end{align*}
Así que la composición de rotaciones centradas en el origen es equivalente a una única rotación centrada en el origen, correspondiente a la suma de los dos ángulos de rotación.

Las matrices de la forma $\left( 
\begin{array}{cc}
\cos \phi & -\sin \phi \\ 
\sin \phi & \cos \phi%
\end{array}%
\right) $ conmutan.

La inversa de $R_{\theta }\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) $ es $R_{-\theta }\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) $, ya que
\[\left( 
\begin{array}{cc}
\cos \theta & -\sin \theta \\ 
\sin \theta & \cos \theta%
\end{array}%
\right) \left( 
\begin{array}{cc}
\cos \left( -\theta \right) & -\sin \left( -\theta \right) \\ 
\sin \left( -\theta \right) & \cos \left( -\theta \right)%
\end{array}%
\right) =\left( 
\begin{array}{cc}
1 & 0 \\ 
0 & 1%
\end{array}%
\right)\]
El conjunto $\left\{ \left( 
\begin{array}{cc}
\cos \theta & -\sin \theta \\ 
\sin \theta & \cos \theta
\end{array}
\right) \right\} $ es un grupo conmutativo con el producto de matrices.

\paragraph{La composición de dos reflexiones es una rotación}

Sea $\alpha $ el ángulo entre las rectas $y=mx$ y $y=nx$.
\begin{align*}
\alpha &= \angle \left( (1, m), (1, n) \right) \\[1ex]
\cos \alpha &= \frac{1 + m n}{\sqrt{m^2 + 1} \sqrt{n^2 + 1}} \\[1ex]
\cos^2 \alpha &= \frac{(1 + m n)^2}{(m^2 + 1)(n^2 + 1)} \\[1ex]
\sin^2 \alpha &= \frac{(m^2 + 1)(n^2 + 1) - (1 + m n)^2}{(m^2+1)(n^2+1)}
= \frac{(m-n)^2}{(m^2 + 1)(n^2 + 1)} \\[1ex]
\cos 2\alpha &= \cos^2\alpha - \sin^2\alpha \\
&= \frac{(1+mn)^2}{(m^2+1)(n^2+1)} - \frac{(m-n)^2}{(m^2+1)(n^2+1)} \\
&= \frac{m^2 n^2 - m^2 + 4mn - n^2 + 1}{(m^2+1)(n^2+1)} \\
&= \frac{(m-n+mn+1)(-m+n+mn+1)}{(m^2+1)(n^2+1)} \\[1ex]
\sin 2\alpha &= 2\sin\alpha \cos\alpha \\
&= 2 \sqrt{ \frac{(m-n)^2}{(m^2+1)(n^2+1)} \cdot \frac{(1+mn)^2}{(m^2+1)(n^2+1)} } \\
&= 2 \frac{ (m-n)(mn+1) }{ (m^2+1)(n^2+1) }
\end{align*}

Así que:
\begin{align*}
T_{m}&\left( T_{n}\left( 
\begin{array}{c}
x \\ 
y%
\end{array}
\right) \right) = \left( 
\begin{array}{cc}
\frac{m^{2}n^{2}-m^{2}+4mn-n^{2}+1}{\left( m^{2}+1 \right)\left( n^{2}+1 \right)} & 
\frac{-2m^{2}n + 2mn^{2} - 2m + 2n}{\left( m^{2}+1 \right)\left( n^{2}+1 \right)} \\[2ex]
\frac{2m^{2}n - 2mn^{2} + 2m - 2n}{\left( m^{2}+1 \right)\left( n^{2}+1 \right)} & 
\frac{m^{2}n^{2}-m^{2}+4mn-n^{2}+1}{\left( m^{2}+1 \right)\left( n^{2}+1 \right)}
\end{array}
\right)
\begin{pmatrix}
x \\ 
y
\end{pmatrix} \\[2ex]
&= \left( 
\begin{array}{cc}
\frac{(m-n+mn+1)(-m+n+mn+1)}{(n^{2}+1)(m^{2}+1)} & 
\frac{-2(m-n)(mn+1)}{(n^{2}+1)(m^{2}+1)} \\[2ex]
\frac{2(m-n)(mn+1)}{(n^{2}+1)(m^{2}+1)} & 
\frac{(m-n+mn+1)(-m+n+mn+1)}{(n^{2}+1)(m^{2}+1)}
\end{array}
\right)
\begin{pmatrix}
x \\
y
\end{pmatrix} \\[2ex]
&= \left( 
\begin{array}{cc}
\cos 2\alpha & -\sin 2\alpha \\ 
\sin 2\alpha & \cos 2\alpha
\end{array}
\right)
\begin{pmatrix}
x \\
y
\end{pmatrix} \\
&= R_{2\alpha}
\begin{pmatrix}
x \\
y
\end{pmatrix}
\end{align*}
En conclusión, la composición de dos reflexiones es una rotación de ángulo $2\alpha $, donde $\alpha$ es el ángulo formado por las rectas de reflexión.

\paragraph{La inversa de una reflexión es ella misma}

La inversa de la reflexión respecto a la recta $y=mx$, $T_{m}$ es ella misma, pues al componerlas nos da la identidad:
\begin{align*}
(T_{m} & \circ T_{m}) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}
\right) \\
&= \left( \frac{1}{m^{2}+1}
\begin{array}{cc}
1-m^{2} & 2m \\ 
2m & m^{2}-1
\end{array}
\frac{1}{m^{2}+1}
\begin{array}{cc}
1-m^{2} & 2m \\ 
2m & m^{2}-1
\end{array}
\right)
\left(
\begin{array}{c}
x \\
y
\end{array}
\right) \\
&= 
\left(
\begin{array}{cc}
1 & 0 \\
0 & 1
\end{array}
\right)
\left(
\begin{array}{c}
x \\
y
\end{array}
\right)
= 
\left(
\begin{array}{c}
x \\
y
\end{array}
\right)
\end{align*}

\begin{example}
\begin{enumerate}
\renewcommand{\theenumi}{\alph{enumi}} \renewcommand{\labelenumi}{(\alph{enumi})}
    \item Halle la ecuación de la curva obtenida al reflejar la curva $y=-x^{2}$; primero, respecto a la recta $y=0$ y luego respecto a la recta $y=x$.
    \item Halle la ecuación de la curva obtenida al rotar la curva $y=-x^{2}\,$un ángulo de $\frac{\pi }{2}$.
\end{enumerate}
\end{example}

\textbf{Solución}

\begin{enumerate}[label=(\alph*)]
\item Al reflejar la curva respecto al eje $x$, reemplazamos $y$ por $-y$ en $y=-x^{2}$ y obtenemos $y=x^{2}$. Luego, al reflejar la curva respecto a la recta $y=x$, intercambiamos las variables $x$ y $y$ en $y=x^{2}$, obteniendo la ecuación $x=y^{2}$.

\item Al rotar un ángulo de $\frac{\pi }{2}$, usamos la matriz de rotación inversa:

\subsubsection{Dilataciones}

Podemos usar un factor de escala para hacer una copia más grande o más pequeña de una figura que también es semejante a la figura original.

Una dilatación con centro en $O$ y factor de escala $c>0$ es una transformación que satisface las siguientes propiedades:

\begin{enumerate}
    \item El centro $O$ es invariante en la dilatación; esto es $O^{\prime }=O$ $\left( T\left( O\right) =O\right)$.
    \item Para cada punto $A\neq O$, $A^{\prime }$ está sobre el segmento de recta que une $O$ con $A$ y cumple que $OA^{\prime }=cOA,$ $\ d\left( 0,T\left( A\right) \right) =cd\left(O,A\right)$.
\end{enumerate}

\enlargethispage*{\baselineskip}

Así que el factor de escala de la dilatación es: 
\[c=\frac{OA^{\prime }}{OA}\]
Y la imagen de una figura bajo una dilatación es semejante a la preimagen (figura~\ref{fig:placeholder3.24}).

\begin{figure}[H]
    \centering
    \caption{Dilatación con centro en $O$}
    \includegraphics[width=0.6\textwidth]{Images/SDY12T8R.png}
    \label{fig:placeholder3.24}
\end{figure}

Una dilatación es un agrandamiento (figura~\ref{fig:placeholder3.25}) si el factor de escala es mayor que 1. Por otro lado, la dilatación es una reducción si el factor de escala está entre 0 y 1. (Nótese que si el factor es 1, la figura permanece invariable).

\begin{figure}[H]
    \centering
    \caption{Dilatación con factor de escala maypr que 1}
    \includegraphics[width=0.6\textwidth]{Images/SDY12T8S.png}
    \label{fig:placeholder3.25}
\end{figure}

La transformación de dilatación centrada en el origen y con factor de escala $c>0$, matricialmente tiene la forma:
\[\left( 
\begin{array}{c}
x^{\prime} \\ 
y^{\prime}%
\end{array}%
\right) =T\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) =\left( 
\begin{array}{c}
cx \\ 
cy%
\end{array}%
\right) =\left( 
\begin{array}{cc}
c & 0 \\ 
0 & c%
\end{array}%
\right) \left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right)\]
La transformación $T$ tiene inversa, por lo que se puede despejar $x$ y $y$ en términos de $x^{\prime}$, $y^{\prime}$:
\[\left( 
\begin{array}{c}
x \\ 
y%
\end{array}%
\right) =\left( 
\begin{array}{c}
\frac{1}{c}x^{\prime} \\ 
\frac{1}{c}y^{\prime}%
\end{array}%
\right) \]

\begin{example}
Halle la imagen de la curva $x^{2}+y^{2}=1$ por una dilatación de factor de escala $c=4$.
\end{example}

\textbf{Solución}

Reemplazamos en la ecuación de la curva $x,y$ por $\frac{1}{4}x^{\prime},\frac{1}{%
4}y^{\prime}$.
\begin{align*}
x^{2}+y^{2}&=1\\
\left( \frac{x^{\prime}}{4}\right) ^{2}+\left( \frac{y^{\prime}}{4}\right) ^{2}&=1 \end{align*}
Multiplicamos por $16$.
\[\left( x^{\prime}\right) ^{2}+\left( y^{\prime}\right) ^{2}=16\]
En la figura~\ref{fig:placeholder3.26}, se dibuja la circunferencia y su imagen dilatada.

\begin{figure}[H]
    \centering
    \caption{Dilatación de una circunferencia}
    \includegraphics[width=0.6\textwidth]{Images/SDY12T8T.png}
    \label{fig:placeholder3.26}
\end{figure}

\paragraph{Toda dilatación preserva ángulos}

Sea T$\left( x,y\right) =c\left( x,y\right) $ una dilatación de factor de escala $c>0.$
\begin{align*}
\cos\angle \left( T\left( u\right) ,T\left( v\right) \right) &=\frac{T\left(
u\right) \cdot T\left( v\right) }{\left\Vert T\left( u\right) \right\Vert
\left\Vert T\left( v\right) \right\Vert } \\ 
&= \frac{cu\cdot cv}{\left\Vert cu\right\Vert \left\Vert cv\right\Vert } \\ 
&= \frac{c^{2}\left( u\cdot v\right) }{c^{2}\left\Vert u\right\Vert \left\Vert v\right\Vert }=\frac{\left( u\cdot v\right) }{\left\Vert u\right\Vert \left\Vert v\right\Vert } \\ 
&=\cos\angle\left( T\left( u\right) ,T\left( v\right) \right)
\end{align*}
Por tanto $\angle \left( Ta\left( u\right) ,T\left( v\right) \right) =\angle
\left( u,v\right) $
